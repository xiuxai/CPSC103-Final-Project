    """

    #return []  #stub

    with open(filename) as csvfile:
        
        reader = csv.reader(csvfile)

        for row in reader:
            # you may not need to store all the rows, and you may need

@typecheck
def is_reliable(row: List[str]) -> bool:
    """
    Return True if the GII rank (row[0]), female population with secondary education (row[8]) 
    and male popoulation with secondary education (row[9]) is not empty.
    
    Assumes row is a full list of values from data dataset.
    Specifically, row[0], row[8] and row[9] must exist.
    
    blank data cells are not reliable since they indicate missing data.
    """
    
    #return True #stub
    
    # template
    
    return row[0] != "" and row[8] != ".." and row[9] != ".."


@typecheck
def analyze(country_data: List[Country]) -> None: 
    """ 
    display a scatterplot of countries based on the following info: GII rank vs. average 
    percentage of population with secondary education(both men and women)
    """ 
    #return None  #stub
    # Template from visualization
    
    pyplot.title('GII Rank vs. Secondary Education')
    pyplot.ylabel('GII Rank')
    pyplot.xlabel('Average Percentage of Population with Secondary Education')
    pyplot.scatter(get_edu_list(country_data), get_ranking_list(country_data))
    pyplot.show()
    
    return None
    
@typecheck 
def get_ranking_list(loc: List[Country]) -> List[int]:
    '''
    returns the GII ranking of each country in the list of countries
    '''
    #return []  #stub
    #template from List[Country]
    
    # rank is a list of the rankings seen so far
    rank = []   # type: List[int]
    for c in loc:
        rank.append(c.gii_rank)
    
    return rank

@typecheck 
def get_edu_list(loc: List[Country]) -> List[float]:
    '''
    returns the average percentage of population with secondary education
    for each country in the list 
    '''
    #return []  #stub
    #template from List[Country]
    
    # edu_list is a list of the average percent of secondary education seen so far
    edu_list = []   # type: List[float]
    for c in loc:
        edu_list.append(average_of_edu(c))
    
    return edu_list  

@typecheck 
def average_of_edu(c: Country) -> float:
    '''
    returns the average percentage of population with secondary education, 
    rounds to two decimal places
    '''
    #return 0 #stub
    # template from Country
    return round((c.pop_w_edu_f + c.pop_w_edu_m) / 2 , 2)
